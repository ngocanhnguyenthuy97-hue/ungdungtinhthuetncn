// Tax calculation engine for Vietnam PIT (Law 2025) // Progressive tax system with 5 brackets export interface TaxCalculationInput { totalIncome: number; // Annual income in VND socialInsurance: number; // Social insurance contributions healthInsurance: number; // Health insurance unemploymentInsurance: number; // Unemployment insurance dependents: number; // Number of dependents taxWithheld: number; // Tax already withheld by employer regulatoryPeriod: 'before_2025_07_01' | 'after_2025_07_01'; taxYear?: number; // Năm quyết toán (2025, 2026, ...) } export interface TaxCalculationResult { grossIncome: number; insuranceContributions: number; personalDeduction: number; dependentDeduction: number; taxableIncome: number; taxInBrackets: Array<{ bracket: number; rate: number; lower: number; upper: number; amount: number; tax: number; }>; totalTax: number; taxPaid: number; refundAmount: number; additionalTax: number; taxYear: number; } // Legacy brackets up to 30/06/2025 (7 bậc) const OLD_TAX_BRACKETS = [ { rate: 0.05, lower: 0, upper: 5_000_000 }, { rate: 0.1, lower: 5_000_000, upper: 10_000_000 }, { rate: 0.15, lower: 10_000_000, upper: 18_000_000 }, { rate: 0.2, lower: 18_000_000, upper: 32_000_000 }, { rate: 0.25, lower: 32_000_000, upper: 52_000_000 }, { rate: 0.3, lower: 52_000_000, upper: 80_000_000 }, { rate: 0.35, lower: 80_000_000, upper: Infinity }, ]; // 5 bậc từ 01/07/2025 và 01/01/2026 cho thu nhập lương, tiền công const NEW_TAX_BRACKETS = [ { rate: 0.05, lower: 0, upper: 10_000_000 }, { rate: 0.1, lower: 10_000_000, upper: 30_000_000 }, { rate: 0.2, lower: 30_000_000, upper: 60_000_000 }, { rate: 0.3, lower: 60_000_000, upper: 100_000_000 }, { rate: 0.35, lower: 100_000_000, upper: Infinity }, ]; // Giảm trừ gia cảnh theo năm áp dụng const PERSONAL_DEDUCTION_2025 = 11_000_000; // per month const DEPENDENT_DEDUCTION_2025 = 4_400_000; // per month per dependent const PERSONAL_DEDUCTION_2026 = 15_500_000; // per month const DEPENDENT_DEDUCTION_2026 = 6_200_000; // per month per dependent export function calculateTax(input: TaxCalculationInput): TaxCalculationResult { const year = input.taxYear || (input.regulatoryPeriod === 'before_2025_07_01' ? 2024 : 2026); const personalDeductionMonthly = year >= 2026 ? PERSONAL_DEDUCTION_2026 : PERSONAL_DEDUCTION_2025; const dependentDeductionMonthly = year >= 2026 ? DEPENDENT_DEDUCTION_2026 : DEPENDENT_DEDUCTION_2025; // Calculate monthly values (convert annual to monthly rates) const monthlyGrossIncome = input.totalIncome / 12; const monthlyInsurance = input.socialInsurance / 12 + input.healthInsurance / 12 + input.unemploymentInsurance / 12; const monthlyPersonalDeduction = personalDeductionMonthly; const monthlyDependentDeduction = dependentDeductionMonthly * input.dependents; // Calculate taxable income for monthly const monthlyTaxableIncome = Math.max( 0, monthlyGrossIncome - monthlyInsurance - monthlyPersonalDeduction - monthlyDependentDeduction ); // Calculate taxable income for annual const taxableIncome = monthlyTaxableIncome * 12; // Select bracket set by period/year const taxBrackets = year <= 2024 || input.regulatoryPeriod === 'before_2025_07_01' ? OLD_TAX_BRACKETS : NEW_TAX_BRACKETS; let totalTax = 0; const taxInBrackets: TaxCalculationResult['taxInBrackets'] = []; for (let i = 0; i < taxBrackets.length; i++) { const bracket = taxBrackets[i]; const nextBracketLower = i < taxBrackets.length - 1 ? taxBrackets[i + 1].lower : bracket.upper; let bracketIncome = 0; if (monthlyTaxableIncome > bracket.lower) { const upper = i === taxBrackets.length - 1 ? monthlyTaxableIncome : Math.min(monthlyTaxableIncome, nextBracketLower); bracketIncome = Math.max(0, upper - bracket.lower); } const bracketTax = bracketIncome * bracket.rate; totalTax += bracketTax; taxInBrackets.push({ bracket: i + 1, rate: bracket.rate * 100, lower: bracket.lower, upper: bracket.upper === Infinity ? undefined : bracket.upper, amount: bracketIncome, tax: bracketTax, }); } // Annual tax (monthly tax * 12) const annualTax = totalTax * 12; // Calculate refund or additional tax const refundAmount = input.taxWithheld > annualTax ? input.taxWithheld - annualTax : 0; const additionalTax = annualTax > input.taxWithheld ? annualTax - input.taxWithheld : 0; return { grossIncome: input.totalIncome, insuranceContributions: input.socialInsurance + input.healthInsurance + input.unemploymentInsurance, personalDeduction: personalDeductionMonthly * 12, dependentDeduction: dependentDeductionMonthly * 12, taxableIncome: Math.round(taxableIncome), taxInBrackets: taxInBrackets.map(b => ({ ...b, amount: Math.round(b.amount), tax: Math.round(b.tax), })), totalTax: Math.round(annualTax), taxPaid: input.taxWithheld, refundAmount: Math.round(refundAmount), additionalTax: Math.round(additionalTax), taxYear: year, }; } export function checkEligibility( companiesCount: number, employerFinalizedTax: boolean, hasAdditionalIncome: boolean ): { needsFinalization: boolean; reason: string } { // Logic: Need to finalize if: // - Multiple employers // - Employer didn't finalize // - Additional income sources if (companiesCount > 1) { return { needsFinalization: true, reason: 'Bạn có thu nhập từ nhiều công ty, cần tự quyết toán thuế', }; } if (!employerFinalizedTax) { return { needsFinalization: true, reason: 'Công ty không quyết toán thuế, bạn cần tự quyết toán', }; } if (hasAdditionalIncome) { return { needsFinalization: true, reason: 'Bạn có thu nhập phát sinh khác, cần tự quyết toán thuế', }; } return { needsFinalization: false, reason: 'Công ty đã quyết toán thuế cho bạn, không cần tự quyết toán', }; } export function formatCurrency(value: number): string { return new Intl.NumberFormat('vi-VN', { style: 'currency', currency: 'VND', minimumFractionDigits: 0, maximumFractionDigits: 0, }).format(value); } export function formatNumber(value: number): string { return new Intl.NumberFormat('vi-VN').format(value); }
